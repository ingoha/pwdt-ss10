###############################################################################
# grmfetchStr : variation on grmfetch, returns a list of strings 
#               instead of a list of datum handles.
#
proc grmfetchStr { grmId field } {
   if [catch {set dhList [grmfetch $grmId ${field}]} err] {
     error $err
   }
   set fieldvalue {}
   foreach dh $dhList {
     lappend fieldvalue [datget $dh VALUE]
     datdestroy $dh
   }   
   return $fieldvalue
}
